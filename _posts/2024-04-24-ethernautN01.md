# 靶场第一关

**要求：获得合约的使用权并使用withdraw函数调回ETH**

```solidity
// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract Fallback {
    mapping(address => uint256) public contributions;
    address public owner;

    constructor() {
        owner = msg.sender;
        contributions[msg.sender] = 1000 * (1 ether);
    }

    modifier onlyOwner() {
        require(msg.sender == owner, "caller is not the owner");
        _;
    }

    function contribute() public payable {
        require(msg.value < 0.001 ether);
        contributions[msg.sender] += msg.value;
        if (contributions[msg.sender] > contributions[owner]) {
            owner = msg.sender;
        }
    }//通过这个函数可以向合约发送ETH，获得贡献值

    function getContribution() public view returns (uint256) {
        return contributions[msg.sender];
    }

    function withdraw() public onlyOwner//要想执行这个函数必须获得合约的所有权  {
        payable(owner).transfer(address(this).balance);将合约中的ETH发送给所有者
    }

    receive() external payable {
        require(msg.value > 0 && contributions[msg.sender] > 0);
        owner = msg.sender;
    }
}

```

总结：先调用contribute发送eth获得contribution，然后再调用receive函数获得合约所有权，再通过调用withdraw函数返回eth
